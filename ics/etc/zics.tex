\def\makeheadline{
    \vbox to 0pt{
	\vskip-102pt
	\line{\vbox to 20pt{}\hrulefill}
        \line{\vbox to 40pt{}\fS \hss\TitleLine\hss}
	\line{\vbox to 20pt{}\hrulefill}
	\vss
    }
    \nointerlineskip
}
